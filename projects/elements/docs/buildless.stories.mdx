# Buildless Integration

Using [Import Maps](https://web.dev/import-maps-in-all-modern-browsers/) it is possible to
use elements without build tooling such as Vite or Rollup for easy in browser **prototyping**.

[Demo Project](https://git-av.nvidia.com/r/plugins/gitiles/elements/+/refs/heads/main/src/ui/platform/design-system/demos/buildless/)

```html
<!DOCTYPE html>
<html lang="en" mlv-theme="dark">
  <head>
    <title>Import Map Starter + @elements/elements</title>
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <link rel="stylesheet" type="text/css" href="./node_modules/@elements/elements/dist/index.css" />
    <script type="importmap">
      {
        "imports": {
          "@elements/elements": "./node_modules/@elements/elements/dist/",
          "@elements/elements/": "./node_modules/@elements/elements/dist/",
          "composed-offset-position": "./node_modules/composed-offset-position/dist/composed-offset-position.esm.js",
          "lit": "./node_modules/lit/index.js",
          "lit/": "./node_modules/lit/",
          "lit-html": "./node_modules/lit-html/lit-html.js",
          "lit-html/": "./node_modules/lit-html/",
          "lit-element/": "./node_modules/lit-element/",
          "@lit/reactive-element": "./node_modules/@lit/reactive-element/reactive-element.js",
          "@lit/reactive-element/": "./node_modules/@lit/reactive-element/",
          "@lit-labs/motion": "./node_modules/@lit-labs/motion/index.js",
          "@floating-ui/dom": "./node_modules/@floating-ui/dom/dist/floating-ui.dom.esm.js",
          "@floating-ui/core": "./node_modules/@floating-ui/core/dist/floating-ui.core.esm.js",
          "@elements/elements/": "./node_modules/@elements/elements/dist/"
        }
      }
    </script>
  </head>
  <body mlv-text="body" mlv-layout="column gap:md pad:lg">
    <mlv-alert status="success">hello world</mlv-alert>

    <script type="module">
      import '@elements/elements/alert/define.js';
    </script>
  </body>
</html>
```

<html mlv-theme="light">
<head>
 <link rel="stylesheet" type="text/css" href="https://cdn-stage.nvidia.com/assets/elements-playground/modules/@elements/elements@0.11.2/dist/index.css" />
  <link rel="stylesheet" type="text/css" href="https://cdn-stage.nvidia.com/assets/elements-playground/modules/@elements/elements@0.11.2/dist/inter.css" />

    {
      "imports": {
        "@elements/elements": "./node_modules/@elements/elements/dist/",
        "@elements/elements/": "./node_modules/@elements/elements/dist/"
        "composed-offset-position": "./node_modules/composed-offset-position/dist/composed-offset-position.esm.js",
        "lit": "./node_modules/lit/index.js",
        "lit/": "./node_modules/lit/",
        "lit-html": "./node_modules/lit-html/lit-html.js",
        "lit-html/": "./node_modules/lit-html/",
        "lit-element/": "./node_modules/lit-element/",
        "@lit/reactive-element": "./node_modules/@lit/reactive-element/reactive-element.js",
        "@lit/reactive-element/": "./node_modules/@lit/reactive-element/",
        "@lit-labs/motion": "./node_modules/@lit-labs/motion/index.js",
        "@floating-ui/dom": "./node_modules/@floating-ui/dom/dist/floating-ui.dom.esm.js",
        "@floating-ui/core": "./node_modules/@floating-ui/core/dist/floating-ui.core.esm.js",
        "@elements/elements/": "./node_modules/@elements/elements/dist/"
      }
    }
  </script>
  <script type="module" src="./index.js"></script>
</head>
  <body mlv-text="body" mlv-layout="column gap:md pad:lg">
    <app-status-label status="success"></app-status-label>

    <script type="module">
      import { html, LitElement } from 'lit';
      import typography from '@elements/elements/css/module.typography.css' assert { type: 'css' };
      import layout from '@elements/elements/css/module.layout.css' assert { type: 'css' };
      import '@elements/elements/badge/define.js';

      class AppStatusLabel extends LitElement {
        static styles = [typography, layout];

        static properties = {
          status: { type: String }
        };

        render() {
          return html`
            <div mlv-layout="row gap:sm align:center">
              <p mlv-text="label">status:</p>
              <mlv-badge .status=${this.status}>${this.status}</mlv-badge>
            </div>
          `;
        }
      }

      customElements.define('app-status-label', AppStatusLabel);
    </script>
  </body>
</html>
